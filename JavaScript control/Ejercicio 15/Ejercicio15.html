Código JavaScript::

<script type="text/javascript">

var contador, numero, num, mayor;
contador = 1;
numero = window.prompt("Ingresa un número entero de un solo dígito:", "0");
num = parseInt(numero);
mayor = num;
contador++;
while(contador <= 10){
numero = window.prompt("Ingresa un número entero de un solo dígito:", "0");
num = parseInt(numero);
if(num > mayor){
mayor = num;
}
contador++;
}
document.writeln("<h3>El número más grande es: "+ mayor +"</h3>");

</script>
