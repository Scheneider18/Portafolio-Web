Pseudocódigo:

Ingresar valor del articulo
Convertir cadena a float
Sumar en un total los valores de los articulos
Mostrar los valores de los articulos vendidos
Calcular el ingreso semanal del vendedor
Mostrar el ingreso del vendedor

Código JavaScript::

<script type="text/javascript">

var articulo, valor, val, venta, ingreso;
articulo = 1;
venta = 0;
valor = window.prompt("Ingrese el valor del articulo vendido, ingrese -1 para salir:", "0");
val = parseFloat(valor);
while(val != -1.00){
venta += val;
document.writeln("<p>Articulo: "+ articulo +", valor: "+ val +"</p>")
articulo++;
valor = window.prompt("Ingrese el valor del articulo vendido, ingrese -1 para salir:", "0");
val = parseFloat(valor);
}
ingreso = (venta * 0.09) + 200;
document.writeln("<h3>Ingreso semanal del vendedor: "+ ingreso + "</h3>");

</script>
