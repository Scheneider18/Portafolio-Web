Pseudocódigo:

Ingresar dato de km recorridos
Verificar dato de entrada
Entrar al bucle
Solicitar litros
Convertir a enteros
Realizar la operación
Imprimir datos en HTML5
Volver a solicitar datos
Salir del bucle
Realizar la operación total Imprimir total en HTML5

Código JavaScript:

<script type="text/javascript">

var litros, l, kilometraje, km, lxkm, totalkm, totallt, viaje;
totalkm = 0;
totallt = 0;
viaje = 1;
kilometraje = window.prompt("Escriba los kilometros recorridos del viaje (enteros), escriba -1 para salir: ", "0");
km = parseInt(kilometraje);
while(km != -1){
totalkm += km;
litros = window.prompt("Escriba los litros consumidos del viaje (enteros):","0");
l = parseInt(litros)
lxkm = km / l;
totallt += l;
document.writeln("<p>En el viaje "+ viaje + " se obtuvo "+ lxkm +" kilometros recorridos por cada litro. </p>")
viaje++;
kilometraje = window.prompt("Escriba los kilometros recorridos del viaje (enteros), escriba -1 para salir: ", "0");
km = parseInt(kilometraje);
}
lxkm = totalkm / totallt;
document.writeln("<p>El total de kilometros recorridos por litro es "+ lxkm +"</p>")

</script>
