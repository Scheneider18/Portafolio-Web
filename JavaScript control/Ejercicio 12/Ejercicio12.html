Pseudocódigo:

Ingresar datos solicitados
Convertir en enteros
Calcular el nuevo balance
Imprimir nuevo balance
Comparar si el nuevo balance es mayor al crédito
Si es cierto imprimir texto diciendo
En caso contrario no hacer nada

Código JavaScript::

<script type="text/javascript">

var nomcu, salini, ini, cargados, car, credito, cred, limite, lim, balance, permitido;
nomcu = window.prompt("Ingrese el número de cuenta:");
salini = window.prompt("Ingrese el saldo inicial en enteros:", "0");
cargados = window.prompt("Ingrese el total de articulos cargados en enteros:","0");
credito = window.prompt("Ingrese el total de creditos aplicados en enteros:","0");
limite = window.prompt("Ingrese el límite de crédito permitido en entero:","0");
ini = parseInt(salini);
car = parseInt(cargados);
cred = parseInt(credito);
lim = parseInt(limite);
balance = ini + car - cred;
document.writeln("<h2>Nuevo balance de: "+ balance +"</h2>")
if(balance > lim){
document.writeln("<p><strong>Se excedió el límite de su crédito.</strong></p>")
}

</script>
