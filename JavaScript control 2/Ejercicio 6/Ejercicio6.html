Ejercicios de autoevaluación: 8.6

Código JavaScript:

<script>

var cuenta, numero, num, menor, x;
x = 1;
cuenta = window.prompt("Ingrese el total de enteros a introducir:");
do{
numero = window.prompt("Ingrese el entero:");
num = parseInt(numero);
if(menor != null){
if(num < menor){
menor = num;
}
}else{
menor = num;
}
++x;
} while ( x <= cuenta);
document.writeln("<h3>El menor entero no negativo introducido es: " + menor +"</h3>");

</script>
