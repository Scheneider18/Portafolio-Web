6.22 Escriba una secuencia de comandos que lea dos enteros, determine y produzca texto de HTML5 que muestre si el primero es múltiplo del segundo. [Sugerencia: use el operador residuo.]

<script type="text/javascript">

var primerNumero, segundoNumero, numero1, numero2;

primerNumero = window.prompt("Escriba el primer entero: ");

segundoNumero = window.prompt("Escriba el segundo entero: ");

numero1 = parseInt(primerNumero);

numero2 = parseInt(segundoNumero);

if (numero1 % numero2 == 0){

document.writeln("<h1> El numero " + numero1 + "es multiplo de" + numero2 + "</h1>");

}else{

document.writeln("<h1> El numero " + numero1 + " no es multiplo de" + numero2 + "</h1>");

}
