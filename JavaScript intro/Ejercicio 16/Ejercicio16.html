Ejercicios de autoevaluación: 6.16

6.16 Escriba una secuencia de comandos que pida al usuario que escriba dos números, obtenga los dos números del usuario y produzca texto que muestre la suma, producto, diferencia, cociente. Use las técnicas que se muestran en la figura 6.7.

<script type="text/javascript">

var primerNumero;

var segundoNumero;

var numero1;

var numero2;

var suma;

var producto;

var diferencia;

var cociente;

primerNumero = window.prompt("Escriba el primer entero");

segundoNumero = window.prompt("Escriba el segundo entero");

numero1 = parseInt(primerNumero);

numero2 = parseInt(segundoNumero);

suma = numero1 + numero2;

producto = numero1 * numero2;

diferencia = numero1 - numero2;

cociente = numero1 / numero2;

document.writeln("La suma de tus números es: " + suma);

document.writeln("El producto de tus números es: " + producto);

document.writeln("La diferencia de tus números es: " + diferencia);

document.writeln("El cociente de tus números es: " + cociente);

</script>
