6.23 Escriba una secuencia de comandos que reciba tres números, determine y produzca marcado que muestre la cantidad de números negativos, positivos y ceros introducidos.

<script>

var primerNumero, segundoNumero, tercerNumero, numero1, numero2, numero3, neg, pos, cero;

primerNumero = window.prompt("Escriba el primer entero: ");

segundoNumero = window.prompt("Escriba el segundo entero: ");

tercerNumero = window.prompt("Escriba el tercer entero: ");

numero1 = parseInt(primerNumero);

numero2 = parseInt(segundoNumero);

numero3 = parseInt(tercerNumero);

neg = 0;

pos = 0;

cero = 0;

if(numero1 == 0){

cero = cero + 1;

}

if(numero1 < 0){

neg = neg + 1;

}

if(numero1 > 0){

pos = pos + 1;

}

if(numero2 == 0){

cero = cero + 1;

}

if(numero2 < 0){

neg = neg + 1;

}

if(numero2 > 0){

pos = pos + 1;

}

if(numero3 == 0){

cero = cero + 1;

}

if(numero3 < 0){

neg = neg + 1;

}

if(numero3 > 0){

pos = pos + 1;

}

document.writeln("<h1>El total de numeros negativos introducidos son :" + neg + "</h1>");

document.writeln("<h1>El total de numeros positivos introducidos son :" + pos + "</h1>");

document.writeln("<h1>El total de numeros ceros introducidos son :" + cero + "</h1>");

</script>
