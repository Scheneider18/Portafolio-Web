6.19 Escriba una secuencia de comandos que obtenga del usuario el radio de un círculo y produzca texto HTML5 que muestre el diámetro, la circunferencia y el área de ese círculo. Use el valor constante 3.14159 para π. Use las técnicas de GUI que se muestran en la figura 6.7. Use las siguientes fórmulas (r es el radio): diametro = 2r, circunferencia = 2πr, área = πr2.

<script>

var radioUser;

var radio;

var diametro;

var circunferencia;

var area;

var pi;

pi = 3.14159;

radioUser = window.prompt("Escriba el radio del círculo: ");

radio = parseFloat(radioUser);

diametro = 2 * radio;

document.writeln("El diametro del circulo es: " + diametro);

circunferencia = 2 * pi * radio;

document.writeln("EL circunferencia es: " + circunferencia);

area = pi * (radio * radio);

document.writeln("El área es: " + area);

</script>
